function readParagraphs(){
  let paragraphs =
    document.querySelectorAll('p');
  paragraphs.forEach( (it) => {
    let text = it.innerText;
    let u =
      new SpeechSynthesisUtterance(text);
    window.speechSynthesis.speak(u);
  });
}
